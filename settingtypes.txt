# enables stoneworks arches
arches (Stoneworks' Arches) bool true

# only useful if arches enabled
# enables wooden nodes from Minetest Game default
arches_wood (Wooden Arches) bool true

# only useful if arches enabled
# enables ores nodes from Minetest Game default
arches_ores (Arches out of ores) bool false

# only useful if arches enabled
# enables stone nodes from Minetest Game default
arches_stone (Arches out of stone) bool true

# only useful if arches enabled
# enables nodes from darkage mod
arches_darkage (Arches out of darkage nodes) bool false

# only useful if arches enabled
# enables nodes from bakedclay mod
arches_bakedclay (Arches out of bakedclay nodes) bool false

walls_wood (Wooden Thinwalls) bool true

# only useful if walls enabled
# enables ores nodes from Minetest Game default
walls_ores (Thinwalls out of ores) bool false

# only useful if walls enabled
# enables stone nodes from Minetest Game default
walls_stone (Thinwalls out of stone) bool true

# only useful if walls enabled
# enables nodes from darkage mod
walls_darkage (Thinwalls out of darkage nodes) bool false

# only useful if walls enabled
# enables nodes from bakedclay mod
walls_bakedclay (Thinwalls out of bakedclay nodes) bool false
