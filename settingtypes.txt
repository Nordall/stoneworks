# enables stoneworks arches
arches (Stoneworks' Arches) bool true

# only useful if arches enabled
# enables wooden nodes from Minetest Game default
arches_wood (Wooden Arches) bool true

# only useful if arches enabled
# enables ores nodes from Minetest Game default
arches_ores (Arches out of ores) bool true

# only useful if arches enabled
# enables stone nodes from Minetest Game default
arches_stone (Arches out of stone) bool true

# only useful if arches enabled
# enables nodes from darkage mod
arches_darkage (Arches out of darkage nodes) bool true

# enables stoneworks miniwalls
walls (Stoneworks' Miniwalls) bool true
# enables default nodes from Minetest Game

# only useful if walls enabled
# enables wooden nodes from Minetest Game default
walls_wood (Wooden Miniwalls) bool true

# only useful if walls enabled
# enables ores nodes from Minetest Game default
walls_ores (Miniwalls out of ores) bool true

# only useful if walls enabled
# enables stone nodes from Minetest Game default
walls_stone (Miniwalls out of stone) bool true

# only useful if walls enabled
# enables nodes from darkage mod
walls_darkage (Miniwalls out of darkage nodes) bool true
